cmake_minimum_required(VERSION 3.10)
if(APPLE)
if( NOT DEFINED CMAKE_BUILD_TYPE)
set(CMAKE_BUILD_TYPE Release)
endif()
if(CMAKE_BUILD_TYPE STREQUAL Debug)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -O0")
endif()
if(CMAKE_BUILD_TYPE STREQUAL Release)
set(CMAKE_OSX_ARCHITECTURES x86_64;arm64)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3")
endif()
endif()
set(CMAKE_CXX_STANDARD 20)
project(badvm CXX)
file(GLOB SRCS src/*.cpp)
add_executable(badvm ${SRCS})
target_include_directories(badvm PUBLIC /usr/local/include ${CMAKE_SOURCE_DIR}/include)
target_link_directories(badvm PUBLIC /usr/local/lib ${CMAKE_SOURCE_DIR}/lib)
target_link_libraries(badvm PUBLIC libSDL2.dylib)
